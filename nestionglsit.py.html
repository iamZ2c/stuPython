aml = []
while 1:
    message = input("输入员工信息")
    if message == "":
        print("程序结束")
        break
    messagelist = message.split(",")
    if len(messagelist) != 3:
        print("格式输入错误请重新输入")
        continue
    aml.append(messagelist)
    for mes in aml:
        print("姓名{n}，年龄{a}，薪资{s}".format(n=mes[0], a=mes[1], s=mes[2]))
